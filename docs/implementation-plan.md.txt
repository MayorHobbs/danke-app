## OVERVIEW: VERTICAL SLICE APPROACH

**Philosophy:**
Build complete **end-to-end features** incrementally, not horizontal layers.

**Bad (Horizontal):**
```
‚ùå Week 1: Build all backend
‚ùå Week 2: Build all UI
‚ùå Week 3: Connect them
   Problem: Nothing works until Week 3
```

**Good (Vertical):**
```
‚úÖ Week 1: MVP widget with hardcoded content (WORKS)
‚úÖ Week 2: Dynamic content loading (WORKS)
‚úÖ Week 3: Full iOS widget + app (WORKS)
‚úÖ Week 4: Android port (WORKS)
   Benefit: Always have a working demo
```

**DailyContent.swift:**

```swift
struct DailyContent: Codable, Identifiable {
    let day: Int
    let quote: String
    let quoteSource: String
    let action: String
    let actionDuration: String
    let domain: Domain
    let difficulty: Difficulty
    
    var id: Int { day }
}

enum Domain: String, Codable {
    case consciousness, health, wealth, relationships
    
    var color: String {
        switch self {
        case .consciousness: return "#8B7FB8"
        case .health: return "#7FA66A"
        case .wealth: return "#C9A961"
        case .relationships: return "#B87F8F"
        }
    }
}

enum Difficulty: String, Codable {
    case beginner, intermediate, advanced
}
```

**ContentService.swift:**
```swift
class ContentService {
    static let shared = ContentService()
    private var allContent: [DailyContent] = []
    
    init() {
        loadContent()
    }
    
    func loadContent() {
        guard let url = Bundle.main.url(forResource: "content", withExtension: "json"),
              let data = try? Data(contentsOf: url),
              let content = try? JSONDecoder().decode([DailyContent].self, from: data) else {
            print("Failed to load content")
            return
        }
        allContent = content
    }
    
    func getContent(forDay day: Int) -> DailyContent? {
        // Handle looping (day 8 wraps to day 1 if only 7 days)
        let index = (day - 1) % allContent.count
        return allContent.indices.contains(index) ? allContent[index] : nil
    }
    
    func getCurrentDayContent() -> DailyContent {
        let currentDay = UserPreferences.shared.currentDay
        return getContent(forDay: currentDay) ?? allContent[0]
    }
}
```

**UserPreferences.swift (Simple UserDefaults wrapper):**
```swift
class UserPreferences {
    static let shared = UserPreferences()
    
    private let defaults = UserDefaults(suiteName: "group.com.yourusername.danke")!
    
    var currentDay: Int {
        get { defaults.integer(forKey: "currentDay") == 0 ? 1 : defaults.integer(forKey: "currentDay") }
        set { defaults.set(newValue, forKey: "currentDay") }
    }
    
    func incrementDay() {
        currentDay += 1
    }
}
```

```swift
struct Provider: TimelineProvider {
    func getTimeline(in context: Context, completion: @escaping (Timeline<Entry>) -> ()) {
        let currentContent = ContentService.shared.getCurrentDayContent()
        
        // Create entry for right now
        let currentDate = Date()
        let entry = SimpleEntry(date: currentDate, content: currentContent)
        
        // Schedule next refresh at midnight
        var calendar = Calendar.current
        calendar.timeZone = TimeZone.current
        let midnight = calendar.startOfDay(for: calendar.date(byAdding: .day, value: 1, to: currentDate)!)
        
        let timeline = Timeline(entries: [entry], policy: .after(midnight))
        completion(timeline)
    }
}
```

```swift
                        .cornerRadius(8)
                }
                
                // Reflection (appears after completion)
                if isCompleted {
                    VStack(alignment: .leading, spacing: 8) {
                        Text("What did you learn?")
```

**What Users Download:**

**What They Actually Use:**
- Primary: Home screen widget (add via widget gallery)

**Why Not Browser Extension:**
```
‚ùå Chrome/Safari extension can't do:
   - Home screen widgets
   - Lock screen widgets
   - Offline access (always-on visibility)
   - Deep OS integration (midnight refresh, notifications)

‚úÖ Native mobile app can do all of the above
```
